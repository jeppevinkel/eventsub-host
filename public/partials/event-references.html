Channel Ban Event

Event fired when a user is banned from the channel.

{
    event: {
        broadcaster: {
            displayName: string
            id: number
            login: string
        }
        endsAt: number
        isPermanent: boolean
        moderator: {
            displayName: string
            id: number
            login: string
        }
        reason: string
        user: {
            displayName: string
            id: number
            login: string
        }
    }
    nonce: undefined | string
    type: 'CHANNEL.BAN'
}

Channel Points Add Reward Event

Event fired when a reward is added to the channel.

{
    event: {
        backgroundColor: string
        broadcaster: {
            displayName: string
            id: number
            login: string
        }
        cooldownExpiresAt: undefined | number
        cost: number
        defaultImage: {
            url1x: string
            url2x: string
            url4x: string
        }
        globalCooldown: {
            isEnabled: boolean
            seconds: number
        }
        image: undefined | {
            url1x: string
            url2x: string
            url4x: string
        }
        isEnabled: boolean
        isInStock: boolean
        isPaused: boolean
        isUserInputRequired: boolean
        maxPerStream: {
            isEnabled: boolean
            value: number
        }
        maxPerUserPerStream: {
            isEnabled: boolean
            value: number
        }
        prompt: string
        redemptionsRedeemedCurrentStream: undefined | number
        rewardId: string
        shouldRedemptionsSkipRequestQueue: boolean
        title: string
    }
    nonce: undefined | string
    type: 'CHANNEL.REWARD.ADD'
}
